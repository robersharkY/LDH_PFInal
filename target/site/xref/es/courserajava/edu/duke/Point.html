4   /**
5    * The <code>Point</code> class represents a two-dimensional location, constructed from (x,y) 
6    * coordinates with some methods for access and the capability to calculate the distance from 
7    * this point to another point.
8    * 
9    * <P>
10   * Example usage:
11   * 
12   * <PRE>
13   * Point a = new Point(3, 4);
14   * Point b = new Point(2, 5);
15   * double dist = a.distance(b);
16   * </PRE>
17   * 
18   * <P>
19   * This software is licensed with an Apache 2 license, see
20   * http://www.apache.org/licenses/LICENSE-2.0 for details.
21   * 
22   * @author Duke Coursera Team
23   */
24  public class Point {
25      private int x;
26      private int y;
27  
28      /**
29       * Create a <code>Point</code> object from x and y coordinates.
30       * 
31       * @param startx is the x-coordinate
32       * @param starty is the y-coordinate
33       */
34      public Point (int startx, int starty) {
35          x = startx;
36          y = starty;
37      }
38  
39      /**
40       * Returns the x coordinate of this point.
41       * 
42       * @return x coordinate
43       */
44      public int getX () {
45          return x;
46      }
47  
48      /**
49       * Returns the y coordinate of this point.
50       * 
51       * @return y coordinate
52       */
53      public int getY () {
54          return y;
55      }
56  
57      /**
58       * Calculate and return the Euclidean distance from this point to another point.
59       * 
60       * @param otherPt the other point to which distance is calculated
61       * @return the distance from this point to otherPt
62       */
63      public double distance (Point otherPt) {
64          int dx = x - otherPt.getX();
65          int dy = y - otherPt.getY();
66          return Math.sqrt(dx * dx + dy * dy);
67      }
68      
69      /**
70       * Returns a string representation of this point.
71       * 
72       * @return (x,y) for this point.
73       */
74      public String toString(){
75          return "("+x+","+y+")";
76      }
77  
78  
79      // for testing only!
80      public static void main (String[] args) {
81          Point p1 = new Point(3, 4);
82          Point p2 = new Point(6, 8);
83          System.out.println(p1.distance(p2));
84      }
85  }

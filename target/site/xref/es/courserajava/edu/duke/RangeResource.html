6   /**
7    * The <code>RangeResource</code> class represents a range of integer (or whole) numbers and allows
8    * access to these numbers using the method <code>sequence</code>. The integers can then be iterated
9    * over using a <code>for</code> loop
10   * 
11   * <P>
12   * Example usage:
13   * 
14   * <PRE>
15   * // this prints the square number sequence for the first 9 numbers:
16   * //   1, 4, 9, 16, 25, 36, 49, 64, 81 
17   * RangeResource rr = new RangeResource(1, 10);
18   * for (int value : rr.sequence()) {
19   *     System.out.println(value * value);
20   * }
21   * </PRE>
22   *
23   * <P>
24   * Another example usage:
25   * 
26   * <PRE>
27   * // this prints the odd numbers in decreasing order within the given range:
28   * //   37, 35, 33, 31, 29, 27, 25, 23
29   * RangeResource rr = new RangeResource(37, 21, -2);
30   * for (int value : rr.sequence()) {
31   *     System.out.println(value);
32   * }
33   * </PRE>
34   *
35   * <P>
36   * This software is licensed with an Apache 2 license, see
37   * http://www.apache.org/licenses/LICENSE-2.0 for details.
38   * 
39   * @author Duke Software Team
40   */
41  public class RangeResource {

44      private int myIncrement;
45      private List<Integer> myValues;

47      /**
48       * Create a <code>RangeResource</code> object, starting at 0 and going up to but not including
49       * end, [0 - end), that increments by 1.
50       * 
51       * @param end when to stop the range, not included as one of the values
52       * @throws exception if the end is negative
53       */
54      public RangeResource (int end) {
55          this(0, end, 1);
56      }

58      /**
59       * Create a <code>RangeResource</code> object, starting at start and going up to but not
60       * including end, [start - end), that increments by 1.
61       * 
62       * @param start the first value in the range, included as one of the values
63       * @param end when to stop the range, not included as one of the values
64       * @throws exception if the end is less than the start
65       */
66      public RangeResource (int start, int end) {
67          this(start, end, 1);
68      }
69  
70      /**
71       * Create a <code>RangeResource</code> object, starting at start and going up to but not
72       * including end, [start - end), that increments by the amount passed as a parameter.
73       * 
74       * @param start the first value in the range, included as one of the values
75       * @param end when to stop the range, not included as one of the values
76       * @param increment how much to add to get the next value in the range's sequence
77       * @throws exception if increment is negative when the end is greater than the start
78       * @throws exception if increment is positive when the end is less than the start
79       * @throws exception if increment is 0
80       */
81      public RangeResource (int start, int end, int increment) {
82          if (increment == 0) {
83              throw new ResourceException("RangeResource: invalid increment, cannot be 0");
84          }
85          if (end < start && increment > 0) {
86              throw new ResourceException("RangeResource: invalid increment, cannot be positive when end (" + end
87                      + ") is less than start (" + start + ")");
88          }
89          if (end > start && increment < 0) {
90              throw new ResourceException("RangeResource: invalid increment, cannot be negative when end (" + end
91                      + ") is greater than start (" + start + ")");
92          }
93          myStart = start;
94          myEnd = end;
95          myIncrement = increment;
96          myValues = makeValues(start, end, increment);
97      }
98  
99      /**
100      * Create an <code>RangeResource</code> object that is a copy of another range.
101      * 
102      * @param other the original range being copied
103      */
104     public RangeResource (RangeResource other) {
105         this(other.myStart, other.myEnd, other.myIncrement);
106     }

161     // generate the values in the range (much simpler than building an Iterator)
162     private List<Integer> makeValues (int start, int end, int increment) {
163         List<Integer> result = new ArrayList<Integer>();
164         while (true) {
165             if (increment > 0 && start >= end) {
166                 break;
167             }
168             else if (increment < 0 && start <= end) {
169                 break;
170             }
171             result.add(start);
172             start += increment;
173         }
174         return result;
175     }

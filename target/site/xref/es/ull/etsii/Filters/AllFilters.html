1   package es.ull.etsii.Filters;
2   
3   import java.util.ArrayList;
4   
5   /**
6    * Clase AllFilters que implementa la interfaz 'Filter'
7    * Clase que almacena todos los tipos de filtro aplicables a una pelicula
8    */
9   public class AllFilters implements Filter {
10    ArrayList<Filter> filters;
11  
12    /**
13     * Constructor de la clase AllFilters
14     */
15    public AllFilters() {
16      filters = new ArrayList<>();
17    }
18  
19    /**
20     * Método que añade un filtro al vector de interfaz 'Filter'
21     * @param f Filtro a añadir
22     */
23    public void addFilter(Filter f) {
24      filters.add(f);
25    }
26  
27    /**
28     * Método de la interfaz 'Filter' heredado.
29     * Recorre el array de 'Filter' y comprueba que para el identificador que se
30     * le pasa por parametro, el elemento del array devuelve true. Si cualquiera
31     * de los filtros retorna false, el método devuelve false.
32     *
33     * @param id Identificador de la película
34     * @return Boolean True si se cumple con el filtro.
35     */
36    @Override
37    public boolean satisfies(String id) {
38      for (Filter f : filters) {
39        if (!f.satisfies(id)) {
40          return false;
41        }
42      }
43      return true;
44    }
45  }

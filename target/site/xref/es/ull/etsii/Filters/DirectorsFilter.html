5   /**
6    * Clase DirectorsFilter que implementa la interfaz 'Filter'
7    * Clase que con la que filtrar las peliculas mediante su director.
8    */
9   public class DirectorsFilter implements Filter {
10    String directors;
11  
12    /**
13     * Constructor de la clase DirectorsFilter al que se le pasa una string que
14     * contiene el nombre o el apellido de un director
15     * @param directors Nombre del director a filtrar
16     */
17    public DirectorsFilter(String directors) {
18      this.directors = directors;
19    }
20  
21    /**
22     * Método de la interfaz 'Filter' heredado.
23     * Recorre el array de 'Filter' y comprueba que para el identificador que se
24     * le pasa por parametro, el director de esa pelicula es igual al director
25     * del filtro
26     *
27     * @param id Identificador de la película
28     * @return Boolean True si se cumple con el filtro.
29     */
30    @Override
31    public boolean satisfies(String id) {
32      String movieDirectors = MovieDatabase.getDirector(id);
33      String[] filterDirectors = directors.split(",");
34      for (String director : filterDirectors) {
35        if (movieDirectors.contains(director)) {
36          return true;
37        }
38      }
39      return false;
40    }
41  }

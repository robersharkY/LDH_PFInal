1   package es.ull.etsii.Filters;
2   import es.ull.etsii.Movies.MovieDatabase;
3   
4   /**
5    * Clase MinutesFilter que implementa la interfaz 'Filter'
6    * Clase que con la que filtrar las peliculas mediante su duración mínima y
7    * máxima en minutos.
8    */
9   public class MinutesFilter implements Filter {
10    private /*final*/ int minMinutes;
11    private /*final*/ int maxMinutes;
12  
13    private /*final*/ char option;
14  
15    /**
16     * Constructor de la clase MinutesFilter. Se establece como opcion del filtro
17     * la duracion que se encuentra entre dos tiempos en minutos.
18     *
19     * @param minMinutes Mínimo de minutos de duración
20     * @param maxMinutes Máximo de minutos de duración
21     */
22    public MinutesFilter(int minMinutes, int maxMinutes) {
23      this.minMinutes = minMinutes;
24      this.maxMinutes = maxMinutes;
25      option = 'r';
26    }
27  
28    /**
29     * Constructor de la clase MinutesFilter. Se establece como opcion del filtro
30     * la duracion superior al tiempo en minutos pasado por parametro.
31     *
32     * @param minMinutes Mínimo de minutos de duración
33     */
34    // No max minutes given
35    public MinutesFilter(int minMinutes) {
36      this.minMinutes = minMinutes;
37      this.maxMinutes = Integer.MAX_VALUE;
38      option = 'b';
39    }
40  
41    /**
42     * Constructor de la clase MinutesFilter. Se establece como opcion del filtro
43     * la duracion superior o inferior al tiempo en minutos pasado por parametro.
44     *
45     * @param minutes Mínimo de minutos de duración
46     * @param paramOption Opcion elegida. Puede ser maximo de minutos o minimo de minutos
47     */
48    public MinutesFilter(int minutes, char paramOption) {
49      option = paramOption;
50      if (option == 'b') {
51        this.minMinutes = minutes;
52      }
53      if (option == 'l') {
54        this.maxMinutes = minutes;
55      }
56    }
57  
58    public int getMinMinutes(){
59      return minMinutes;
60    }
61  
62    public int getMaxMinutes(){
63      return maxMinutes;
64    }
65  
66    /**
67     * Método de la interfaz 'Filter' heredado.
68     * Comprueba si la(s) duración(es) seleccionada(s) coincide(n) y/u opción
69     * seleccionada coincide con el año de cierta película
70     *
71     * @param id Identificador de la película
72     * @return Boolean True si se cumple con el filtro.
73     */
74    @Override
75    public boolean satisfies(String id) {
76      switch (option) {
77        // Equal
78        case 'r':
79          return ((MovieDatabase.getMinutes(id) >= minMinutes) && (MovieDatabase.getMinutes(id) <= maxMinutes));
80        case 'b':
81          return MovieDatabase.getMinutes(id) >= minMinutes;
82        case 'l':
83          return MovieDatabase.getMinutes(id) <= maxMinutes;
84        default:
85          return false; // Default return
86      }
87    }
88  }

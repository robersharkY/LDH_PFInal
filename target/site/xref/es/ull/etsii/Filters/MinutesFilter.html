5   /**
6    * A class for filter movies by time
7    *
8    * @author Stanislav Rakitov
9    * @version 1.0
10   */
11  public class MinutesFilter implements Filter {
12    private final int minMinutes;
13    private final int maxMinutes;
14  
15    public MinutesFilter(int minMinutes, int maxMinutes) {
16      this.minMinutes = minMinutes;
17      this.maxMinutes = maxMinutes;
18    }
19  
20    // No max minutes given
21    public MinutesFilter(int minMinutes) {
22      this.minMinutes = minMinutes;
23      this.maxMinutes = Integer.MAX_VALUE;
24    }
25  
26    @Override
27    public boolean satisfies(String id) {
28      return MovieDatabase.getMinutes(id) >= minMinutes && MovieDatabase.getMinutes(id) <= maxMinutes;
29    }
30  }

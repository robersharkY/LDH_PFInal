1   package es.ull.etsii.Filters;

3   import es.ull.etsii.Movies.MovieDatabase;
4   
5   /**
6    * Clase GenreFilter que implementa la interfaz 'Filter'
7    * Clase que con la que filtrar las peliculas mediante su género cinematográfico.
8    */
9   public class CountryFilter implements Filter {
10  
11      private final String country;
12  
13      // The constructor should have one parameter named genre representing one genre,
14      // and the satisfies method should return true if a movie has this genre.
15      // Note that movies may have several genres.
16  
17      /**
18       * Constructor de la clase CountryFilter. Se establece el género a
19       * partir del cuál se filtrarán las películas.
20       * @param country Género a filtrar
21       */
22      public CountryFilter(String country) {
23          this.country = country.toLowerCase();
24      }
25  
26      /**
27       * Método de la interfaz 'Filter' heredado.
28       * Comprueba si el genero seleccionado coincide con el conjunto de metodos
29       * que tiene una pelicula
30       *
31       * @param id Identificador de la película
32       * @return Boolean True si se cumple con el filtro.
33       */
34      @Override
35      public boolean satisfies(String id) {
36          return MovieDatabase.getCountry(id).toLowerCase().contains(country);
37      }
38  }

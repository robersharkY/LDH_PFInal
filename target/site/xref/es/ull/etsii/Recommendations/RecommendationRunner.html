11  public class RecommendationRunner implements Recommender {
12    @Override
13    public ArrayList<String> getItemsToRate() {
14      ArrayList<String> movies = MovieDatabase.filterBy(new TrueFilter());
15      ArrayList<String> toRet = new ArrayList<>();
16      for (int i = 0; i < 25; i++) {
17        toRet.add(movies.get(i));
18      }
19      return toRet;
20    }
21  
22    @Override
23    public void printRecommendationsFor(String webRaterID) {
24      FourthRatings fr = new FourthRatings();
25      ArrayList<Rating> movies = fr.getSimilarRatingsByFilter(webRaterID, 50, 2, new TrueFilter());
26      if (movies.size() == 0) {
27        System.out.println("<p>Sorry, there are no recommendations for you.</p>");
28        System.exit(1);
29      }
30      System.out.println("<table>");
31      System.out.println("<tr><th>Rank</th><th>Movie Title</th></tr>");
32      for (int i = 0; i < 15; i++) {
33        System.out.println(
34                "<tr><td>"
35                        + (i + 1)
36                        + "</td><td>"
37                        + MovieDatabase.getTitle(movies.get(i).getItem())
38                        + "</td></tr>");
39      }
40      System.out.println("</table>");
41    }
42  }

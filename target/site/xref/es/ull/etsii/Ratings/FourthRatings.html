23    /**
24     * Método que dado una película, y un número mínimo de valoraciones, devuelve
25     * la nota media de las valoraciones de esa película.
26     *
27     * @param movieID Identificador de la película a buscar
28     * @param minimalRaters Número mínimo de valoraciones que ha de tener la
29     *                      película indicada.
30     * @return Nota media de valoraciones si se cumple el número mínimo de
31     * valoraciones. En caso contrario, devuelve 0.
32     */
33    private Double getAverageByID(String movieID, Integer minimalRaters) {
34      RaterDatabase.initialize("ratings.csv");
35      ArrayList<Rater> myRaters = RaterDatabase.getRaters();
36      long numOfRatings = myRaters.stream().filter(rater -> rater.hasRating(movieID)).count();
37  
38      if (numOfRatings >= minimalRaters) {
39        return myRaters.stream()
40            .filter(rater -> rater.hasRating(movieID))
41            .mapToDouble(rating -> rating.getRating(movieID))
42            .average()
43            .orElse(0.0);
44      }
45      return 0.0;
46    }

48    /**
49     * Método que, dado un número mínimo de valoraciones, devuelve un vector de
50     * Ratings (movie_id,rating_value) con la media de las valoraciones de las
51     * películas que tienen más cantidad de valoraciones que la cantidad de
52     * valoraciones mínimas pasadas por parámetro.
53     *
54     * @param minimalRaters Número mínimo de valoraciones.
55     * @return Lista de películas.
56     */
57    public ArrayList<Rating> getAverageRatings(int minimalRaters) {
58      RaterDatabase.initialize("ratings.csv");
59      ArrayList<Rating> list = new ArrayList<>();
60      ArrayList<String> allMoviesIDs = MovieDatabase.filterBy(new TrueFilter());
61      for (String movieID : allMoviesIDs) {
62        double averageRating = getAverageByID(movieID, minimalRaters);
63        if (averageRating != 0.0) {
64          list.add(new Rating(movieID, averageRating));
65        }
66      }
67      return list;
68    }

70    public ArrayList<Rating> getAverageRatingsByFilter(Integer minimalRaters, Filter filterCriteria) {
71      ArrayList<Rating> ratingsList = new ArrayList<>();
72      ArrayList<String> allMoviesIDs = MovieDatabase.filterBy(filterCriteria);
73      Rating rating;
74      for (String movie_id : allMoviesIDs) {
75        if (getAverageByID(movie_id, minimalRaters) != 0) {
76          rating = new Rating(movie_id, getAverageByID(movie_id, minimalRaters));
77          ratingsList.add(rating);
78        }
79      }
80  
81      sort(ratingsList);
82      return ratingsList;
83    }

85    public double dotProduct(Rater meRater, Rater otherRater) {
86      HashMap<String, Rating> myRatings = meRater.getMyRatings();
87      double result = 0.0;
88      for (String id : myRatings.keySet()) {
89        double scale = meRater.getRating(id) - 5;
90        if (otherRater.hasRating(id)) {
91          result += scale * (otherRater.getRating(id) - 5);
92        }
93      }
94      return result;
95    }

97    private ArrayList<Rating> getSimilarities(String id) {
98      ArrayList<Rating> list = new ArrayList<>();
99      Rater me = RaterDatabase.getRater(id);
100     for (Rater rater : RaterDatabase.getRaters()) {
101       if (!me.getID().equals(rater.getID())) {
102         double dotProduct = dotProduct(me, rater);
103         if (dotProduct > 0) {
104           Rating rating = new Rating(rater.getID(), dotProduct);
105           list.add(rating);
106         }
107       }
108     }
109     list.sort(reverseOrder());
110     return list;
111   }

113   public ArrayList<Rating> getSimilarRatings(
114       String id, Integer numSimilarRaters, Integer minimalRaters) {
115     ArrayList<Rating> ratingMoive = new ArrayList<>();
116     ArrayList<Rating> ratingRater = getSimilarities(id);
117     ArrayList<String> movies = MovieDatabase.filterBy(new TrueFilter());
118     for (String movie_id : movies) {
119       int mdid = Integer.parseInt(movie_id);
120       if (hasMinRaters(movie_id, minimalRaters, numSimilarRaters, ratingRater)) {
121         double sum = 0.0;
122         double ave;
123         double num = 0.0;
124         for (int i = 0; i < numSimilarRaters; i++) {
125           Rater rater = RaterDatabase.getRater(ratingRater.get(i).getItem());
126           HashMap<String, Rating> movieRated = rater.getMyRatings();
127           for (String mo_id : movieRated.keySet()) {
128             int rmid = Integer.parseInt(mo_id);
129             if (rmid == mdid) {
130               sum += ratingRater.get(i).getValue() * rater.getRating(mo_id);
131               num += 1;
132             }
133           }
134         }
135         if (num != 0.0) {
136           ave = sum / num;
137           Rating rating = new Rating(movie_id, ave);
138           ratingMoive.add(rating);
139         }
140       }
141     }
142     ratingMoive.sort(reverseOrder());
143     return ratingMoive;
144   }

146   private boolean hasMinRaters(
147       String movie_id,
148       Integer minimalRaters,
149       Integer numSimilarRaters,
150       ArrayList<Rating> ratingRater) {
151     int numOfRaters = 0;
152     int mdid = Integer.parseInt(movie_id);
153     for (int i = 0; i < numSimilarRaters; i++) {
154       Rater rater = RaterDatabase.getRater(ratingRater.get(i).getItem());
155       HashMap<String, Rating> movieRated = rater.getMyRatings();
156       for (String mo_id : movieRated.keySet()) {
157         int rmid = Integer.parseInt(mo_id);
158         if (rmid == mdid) {
159           numOfRaters += 1;
160         }
161       }
162     }
163     return numOfRaters >= minimalRaters;
164   }

3   import java.util.ArrayList;

5   /**
6    * Implement this interface to allow your code to be integrated with our web site.
7    *
8    * <p>When users first visit the recommender website, our code will call the method <code>
9    * getItemsToRate()</code> to get a list of movies to display on the web page for users to rate.
10   *
11   * <p>When a user submits their ratings, our code will call the method <code>
12   * printRecommendationsFor</code> to get your recommendations based on the user's ratings. The ID
13   * given to this method is for a new Rater that we have already added to the RaterDatabase with
14   * ratings for the movies returned by the first method. Whatever is printed from that method will be
15   * displayed on the web page: HTML, plain text, or debugging information.
16   */
17  public interface Recommender {
18    /**
19     * This method returns a list of movie IDs that will be used to look up the movies in the
20     * MovieDatabase and present them to users to rate.
21     *
22     * <p>The movies returned in the list will be displayed on a web page, so the number you choose
23     * may affect how long the page takes to load and how willing users are to rate the movies. For
24     * example, 10-20 should be fine, 50 or more would be too many.
25     *
26     * <p>There are no restrictions on the method you use to generate this list of movies: the most
27     * recent movies, movies from a specific genre, randomly chosen movies, or simply your favorite
28     * movies.
29     *
30     * <p>The ratings for these movies will make the profile for a new Rater that will be used to
31     * compare to for finding recommendations.
32     */
33    ArrayList<String> getItemsToRate();
34  
35    /**
36     * This method returns nothing, but prints out an HTML table of the movies recommended for the
37     * given rater.
38     *
39     * <p>The HTML printed will be displayed on a web page, so the number you choose to display may
40     * affect how long the page takes to load. For example, you may want to limit the number printed
41     * to only the top 20-50 movies recommended or to movies not rater by the given rater.
42     *
43     * <p>You may also include CSS styling for your table using the &lt;style&gt; tag before you print
44     * the table. There are no restrictions on which movies you print, what order you print them in,
45     * or what information you include about each movie.
46     *
47     * @param webRaterID the ID of a new Rater that has been already added to the RaterDatabase with
48     *     ratings for the movies returned by the method getItemsToRate
49     */
50    void printRecommendationsFor(String webRaterID);
51  }

8   import es.courserajava.edu.duke.FileResource;
9   import es.apachecsv.org.apache.commons.csv.CSVParser;
10  import es.apachecsv.org.apache.commons.csv.CSVRecord;
11  import es.ull.etsii.Rate.Rater;
12  
13  import java.util.ArrayList;
14  import java.util.HashMap;
15  
16  public class RaterDatabase {
17    private static HashMap<String, Rater> ourRaters;
18  
19    private static void initialize() {
20      // this method is only called from addRatings
21      if (ourRaters == null) {
22        ourRaters = new HashMap<String, Rater>();
23      }
24    }
25  
26    public static void initialize(String filename) {
27      if (ourRaters == null) {
28        ourRaters = new HashMap<String, Rater>();
29        addRatings("data/" + filename);
30      }
31    }
32  
33    public static void addRatings(String filename) {
34      initialize();
35      FileResource fr = new FileResource(filename);
36      CSVParser csvp = fr.getCSVParser();
37      for (CSVRecord rec : csvp) {
38        String id = rec.get("rater_id");
39        String item = rec.get("movie_id");
40        String rating = rec.get("rating");
41        addRaterRating(id, item, Double.parseDouble(rating));
42      }
43    }
44  
45    public static void addRaterRating(String raterID, String movieID, double rating) {
46      initialize();
47      Rater rater = null;
48      if (ourRaters.containsKey(raterID)) {
49        rater = ourRaters.get(raterID);
50      } else {
51        rater = new EfficientRater(raterID);
52        ourRaters.put(raterID, rater);
53      }
54      rater.addRating(movieID, rating);
55    }
56  
57    public static Rater getRater(String id) {
58      initialize();
59  
60      return ourRaters.get(id);
61    }
62  
63    public static ArrayList<Rater> getRaters() {
64      initialize();
65      ArrayList<Rater> list = new ArrayList<Rater>(ourRaters.values());
66  
67      return list;
68    }
69  
70    public static int size() {
71      return ourRaters.size();
72    }
73  }
